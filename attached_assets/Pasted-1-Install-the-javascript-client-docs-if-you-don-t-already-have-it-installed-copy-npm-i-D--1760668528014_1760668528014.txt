api_name: /lambda
copy
import { Client } from "@gradio/client";

const client = await Client.connect("Menyu/wainsfw");
const result = await client.predict("/lambda", { 		
		x: true, 
});

console.log(result.data);
Accepts 1 parameter:
x boolean Default: True

The input value that is provided in the "使用反向词条" Checkbox component.

Returns 1 element
string

The output value that appears in the "反向词条" Textbox component.

api_name: /infer
copy
import { Client } from "@gradio/client";

const client = await Client.connect("Menyu/wainsfw");
const result = await client.predict("/infer", { 		
		prompt: "Hello!!", 		
		negative_prompt: "Hello!!", 		
		use_negative_prompt: true, 		
		seed: 0, 		
		width: 512, 		
		height: 512, 		
		guidance_scale: 0.1, 		
		num_inference_steps: 1, 		
		randomize_seed: true, 
});

console.log(result.data);
